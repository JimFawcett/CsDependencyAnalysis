    bool isTerminator(Token tok)
    {
      if (tok == ";" || tok == "{" || tok == "}")
        return true;
      if (tok.Length > 0 && tok[0] == '\n')
      {
        trim();
        if (toks.Count > 0 && toks[0] == "#")
          return true;
      }
      return false;
    }
